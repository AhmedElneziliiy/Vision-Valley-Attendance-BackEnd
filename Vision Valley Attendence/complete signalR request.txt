Code in NotificationHub.cs:
public override async Task OnConnectedAsync()
{
    var userId = Context.User?.FindFirst(ClaimTypes.NameIdentifier)?.Value;
    
    if (!string.IsNullOrEmpty(userId))
    {
        // Add this connection to user-specific group
        await Groups.AddToGroupAsync(Context.ConnectionId, $"user_{userId}");
        _logger.LogInformation("User {UserId} connected to NotificationHub", userId);
    }
    
    await base.OnConnectedAsync();
}

Code in Flutter (Reference):
hubConnection.on('ReceiveNotification', (message) {
  final data = jsonDecode(message[0]);
  
  if (data['type'] == 'LampAccessRequest') {
    // Show notification popup
    showNotification(
      title: 'Lamp Access Request',
      body: '${data["requesterName"]} needs access to ${data["lampName"]}',
      requestId: data['requestId']
    );
  }
});
